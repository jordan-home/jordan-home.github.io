### 冒泡排序
```javascript
function bubbleSort(arr) {
  const len = arr.length
  let needSort = false
  for(let i = 0; i< len; i++) {
    for(let j = 0; j < len - i; j++) {
      const left = arr[j]
      if(left > arr[j+1]) {
        arr[j] = arr[j+1]
        arr[j+1] = left
        needSort = true
      }
    }
    if(!needSort) return arr
  }
  return arr
}
```


### 选择排序
```javascript
function selectSort(arr) {
  const len = arr.length
  for(let i = 0; i < len-1;i++) {
    let minIndex = i
    for(let j = i+1; j < len; j++) {
      if (arr[j] < arr[minIndex]) minIndex = j
    }
    if (minIndex === i) continue
    console.log('continue:', minIndex,i)
    // [arr[minIndex], arr[i]] = [arr[i], arr[minIndex]]
    const minValue = arr[minIndex]
    arr[minIndex] = arr[i]
    arr[i] = minValue
  }
  return arr
}
```


### 插入排序
```javascript
function insertSort(arr) {
  const len = arr.length
  let temp
  for(let i = 1; i<len; i++) {
    let j = i
    temp = arr[i]
    while(j > 0 && arr[j-1] > temp) {
      arr[j] = arr[j-1]
      j--
    }
    arr[j] = temp
  }
  return arr
}
```


### 归并排序
```javascript
function mergeSort(arr) {
  const len = arr.length
  if (len <=1) return arr
  const mid = Math.floor(len/2)
  const leftArr = mergeSort(arr.slice(0,mid))
  const rightArr = mergeSort(arr.slice(mid, len))
  return mergeArr(leftArr, rightArr)
}

function mergeArr(arr1, arr2) {
  const len1 = arr1.length, len2 = arr2.length
  let i = 0, j = 0
  let arr = []
  while(i < len1 && j < len2) {
    if (arr1[i] < arr2[j]) {
      arr.push(arr1[i])
      i++
    } else {
      arr.push(arr2[j])
      j++
    }
  }
  if (i<len1) {
    return arr.concat(arr1.slice(i))
  }else {
    return arr.concat(arr2.slice(j))
  }
}
```


### 快排
```javascript
function quickSort(arr) {
  const len =arr.length
  if (len <= 1) {
    return arr
  }
  const baseIndex = Math.floor(len/2)
  const baseValue = arr.splice(baseIndex, 1)[0]
  let left = []
  let right = []
  for(let i = 0;i<len-1;i++){
    if (arr[i]<baseValue) left.push(arr[i])
    else right.push(arr[i])
  }
  return quickSort(left).concat([baseValue], quickSort(right))
}
```
